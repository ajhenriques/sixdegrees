def degrees_separated_from(other)
  g = Graph.new
  Actor.all.each do |actor|
    g.push actor
  end
  Movie.all.each do |movie|
    movie.actors.each_cons(2) do |ac|
      g.connect_both ac[0], ac[1]
    end
  end
  puts g
  return g.degrees_of_separation(self, other)
end
